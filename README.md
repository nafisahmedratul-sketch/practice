#  এই খানে শুধু কোড দেয়া হবে প্রতিদিন ক্লাস এর।
## প্রথম ক্লাসঃ
```c
#include <stdio.h>

int main() {
    int a = 5;
    int x = 6;        // 3.1416
    float b = 2.54;
    char section = 'b';
    double rabbani = 52.36694694;

    printf("what is your section? =%lf\n", rabbani);
    printf("dge? =%d\n", a);
    printf("what dsfgsf? =%d\n", x);
    printf("what gsdgssection? =%f\n", b);
    printf("what sdfgr section? =%c\n", section);

    return 0;
}

```
## দ্বিতীয় ক্লাস
*loop= for, while, do-while*

*For*
```c
#include<stdio.h>
#include<stdlib.h>

int main()
{
 for (int i=1; i<=10; i++){
        printf("%d\n", i);
        }
    return 0;
}
```


*while*
```c
#include<stdio.h>
#include<stdlib.h>

int main()
{
int i = 1;
        while (i<=10){
            printf("%d\n", i); i++;}
      
      
 return 0;
}

```

*do-while*
```c
#include<stdio.h>
#include<stdlib.h>
int main()
{
 int i = 1;
 do {
        printf("%d\n", i);
        i++;
    }

    while (i<=130);
    return 0;
}

```
